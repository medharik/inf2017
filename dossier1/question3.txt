code sql de création des 2 tables :
 
create table course (
numero_course NUMBER(6),
numero_membre NUMBER(6),
code_type VARCHAR2(15),
date_course DATE,
meilleur_temps DATE ,
CONSTRAINT pk_course PRIMARY KEY (numero_course),
CONSTRAINT fk_membre_course  FOREIGN KEY (numero_membre) 
    REFERENCES membre (numero_membre) ,
CONSTRAINT fk_course_est_type_cou FOREIGN KEY (code_type) 
    REFERENCES type_course (code_type) 
      


)

CREATE TABLE type_course (

code_type  VARCHAR2(15),
identifiant NUMBER(6),
numero_modele NUMBER(6),
tarif FLOAT(5),
nbtours INTEGER, 
CONSTRAINT pk_type_course primary key (code_type) ,
CONSTRAINT fk_circuit_type_course FOREIGN key (identifiant) references circuit(identifiant),

CONSTRAINT fk_modele_type_course FOREIGN key (numero_modele) references modele(numero_modele)



)
